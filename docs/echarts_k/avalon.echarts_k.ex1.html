        <html>
        <head>
            <meta charset='utf-8'/>
            <title>k线图-多级控制</title>
            <script src='../require.js'></script>
            <script>
                require.config({
                    baseUrl:'../../oniui',
                    paths:{
                        avalon:'avalon.shim',
                        echarts: 'echarts',
                        domReady:'domReady'
                    },
                    map:{
                        '*':{
                            'css':'require.css',
                            'text':'require.text'
                        }
                    },
                    shim:{
                        avalon: { exports: "avalon" },
                        echarts:{exports:"echarts"}
                    }
                });
            </script>
        </head>
       <body ms-controller="demo">
            <button ms-widget="button" ms-click="show('ADialog')">点击我获取echarts案例options值</button>
            <div ms-widget="dialog,ADialog">{{optionsStr()}}</div>
            <div id="k" style="width: 800px;height: 300px;"></div>
            <script>
                require(["echarts.fb","echarts/chart/k","dialog/avalon.dialog","button/avalon.button"],function(fb){
                    var vm=avalon.define({
                        $id:"demo",
                        show:function(dialog){
                            avalon.vmodels[dialog].toggle = true;
                        },
                        optionsStr:function(){
                            return JSON.stringify(vm.options);
                        },
                        dialog:{
                            title:"options的json格式"
                        },
                        options:{
                            title : {
                                text: '2013年1月份上证指数'
                            },
                            tooltip : {
                                trigger: 'axis',
                                formatter:function(params){
                                    var seriesName=params[0].seriesName;
                                    var axisName= params[0].name;
                                    var data=params[0].data;
                                    return seriesName+":
"+"时间："+axisName+"
"+
                                            "开盘："+data[0]+"
收盘："+data[1]+
                                            "
最低："+data[2]+"
最高："+data[3]
                                }
                            },
                            legend: {
                                data:[]
                            },
                            xAxis : [
                                {
                                    type : 'category',
                                    data:[]
                                }
                            ],
                            yAxis : [
                                {
                                    type : 'value',
                                    scale:true
                                }
                            ],
                            dataZoom:{
                                show:true,
                                realtime:true,//缩放变化是否实时显示，在不支持Canvas的浏览器中该值自动强制置为false。
                                start:50,
                                end:100
                            },
                            series:[{
                                barMaxWidth: 20,
                                itemStyle: {
                                    normal: {
                                        color: 'red',           // 阳线填充颜色
                                        color0: 'lightgreen',   // 阴线填充颜色
                                        lineStyle: {
                                            width: 2,
                                            color: 'orange',    // 阳线边框颜色
                                            color0: 'green'     // 阴线边框颜色
                                        }
                                    },
                                    emphasis: {
                                        color: 'black',         // 阳线填充颜色
                                        color0: 'white'         // 阴线填充颜色
                                    }
                                },
                                name:"上证指数",
                                type:"k",
                                data:[]
                            }]
                        }
                    });
                    avalon.scan();


                    fb.requestJson({
                        remoteUrl:"./model/k.json",
                        localUrl:"./model/k.json",
                        success:function(data){
                            var options=vm.options;
                            var result=data.rows;
                            var simpleData=[];
                            var xAxisData=[];
                            var seriesData={
                                value:[2320.26,2302.6,2287.3,2362.94],
                                itemStyle: {
                                    normal: {
                                        color0: 'blue',         // 阴线填充颜色
                                        lineStyle: {
                                            width: 3,
                                            color0: 'blue'      // 阴线边框颜色
                                        }
                                    },
                                    emphasis: {
                                        color0: 'blue'          // 阴线填充颜色
                                    }
                                }

                            };
                            options.series[0].data.push(seriesData);
                            for(var i=0;i<result.length;i++){
                                simpleData=[];
                                xAxisData.push(result[i].date);
                                simpleData.push(result[i].openQuotation);
                                simpleData.push(result[i].marketClose);
                                simpleData.push(result[i].min);
                                simpleData.push(result[i].max);
                                options.series[0].data.push(simpleData);
                            }

                            options.xAxis[0].data=xAxisData;
                            var myChart = echarts.init(document.getElementById('k'));
                            myChart.setOption(options);
                        }
                    });
                });
            </script>
        </body>
        </html>

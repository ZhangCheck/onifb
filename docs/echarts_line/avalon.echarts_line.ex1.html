        <!DOCTYPE HTML>
        <html>
        <head>
            <meta charset='utf-8'/>
            <title>标准折线图</title>
            <script src='../require.js'></script>
            <script>
                require.config({
                    baseUrl:'../../oniui',
                    paths:{
                        avalon:'avalon.shim',
                        echarts: 'echarts',
                        domReady:'domReady'
                    },
                    map:{
                        '*':{
                            'css':'require.css',
                            'text':'require.text'
                        }
                    },
                    shim:{
                        avalon: { exports: "avalon" },
                        echarts:{exports:"echarts"}
                    }
                });
            </script>
        </head>
        <body ms-controller="demo">
            <button ms-widget="button" ms-click="show('ADialog')">点击我获取echarts案例options值</button>
            <div ms-widget="dialog,ADialog">{{optionsStr()}}</div>
            <div id="line" style="width: 800px;height: 300px;"></div>
                <script>
                    require(["echarts.fb","echarts/chart/line","dialog/avalon.dialog","button/avalon.button"],function(){
                        var vm=avalon.define({
                            $id:"demo",
                            show:function(dialog){
                                avalon.vmodels[dialog].toggle = true;
                            },
                            optionsStr:function(){
                                return JSON.stringify(vm.options);
                            },
                            dialog:{
                                title:"options的json格式"
                            },
                            options:{
                                title : {
                                    text: '某地区2015年蒸发量和降水量',
                                },
                                tooltip : {
                                    trigger: 'axis'
                                },
                                legend: {
                                    data:[]
                                },
                                calculable : true,
                                grid: {
                                    borderWidth: 0
                                },
                                xAxis : [
                                    {
                                        type : 'category',
                                        splitLine: {
                                            show: false
                                        },
                                        data : []
                                    }
                                ],
                                yAxis : [
                                    {
                                        type : 'value',
                                        splitLine: {
                                            show: false
                                        }
                                    }
                                ],
                                series : [
                                ]
                            }
                        });
                        avalon.scan();

                        fb.requestJson({
                            remoteUrl:"./model/standard_line.json",
                            localUrl:"./model/standard_line.json",
                            data:{},
                            success:function(data){
                                var options=vm.options;
                                var legend=[];//图例
                                var xAxis=[];//x轴数据
                                var data1=[];//蒸发量
                                var data2=[];//降水量
                                var dataObj1={name:"蒸发量",type:"line"};
                                var dataObj2={name:"降水量",type:"line"};
                                var result=data.rows;
                                legend=[data.rows[0].data1.name,data.rows[0].data2.name];
                                options.legend.data=legend;
                                for(var i=0;i<result.length;i++){
                                    xAxis.push(result[i].xAxis);
                                    data1.push(result[i].data1.value);
                                    data2.push(result[i].data2.value);
                                }
                                options.xAxis[0].data=xAxis;
                                dataObj1.data=data1;
                                dataObj2.data=data2;
                                options.series=[dataObj1,dataObj2];
                                var myChart = echarts.init(document.getElementById('line'));
                                myChart.setOption(options);
                            }
                        });
                    });
                </script>
            </div>
        </body>
        </html>

        <head>
            <meta charset='utf-8'/>
            <title>矩形树图-数据下钻</title>
            <script src='../require.js'></script>
            <script>
                require.config({
                    baseUrl:'../../oniui',
                    paths:{
                        avalon:'avalon.shim',
                        echarts: 'echarts',
                        domReady:'domReady'
                    },
                    map:{
                        '*':{
                            'css':'require.css',
                            'text':'require.text'
                        }
                    },
                    shim:{
                        avalon: { exports: "avalon" },
                        echarts:{exports:"echarts"}
                    }
                });
            </script>
        </head>
        <body ms-controller="demo">
            <button ms-widget="button" ms-click="show('ADialog')">点击我获取echarts案例options值</button>
            <div ms-widget="dialog,ADialog">{{optionsStr()}}</div>
            <div id="treemap" style="width: 800px;height: 300px;"></div>
            <script>
               require(["echarts.fb","echarts/chart/treemap","dialog/avalon.dialog","button/avalon.button"],function(fb){
                    var vm=avalon.define({
                        $id:"demo",
                        show:function(dialog){
                            avalon.vmodels[dialog].toggle = true;
                        },
                        optionsStr:function(){
                            return JSON.stringify(vm.options);
                        },
                        dialog:{
                            title:"options的json格式"
                        },
                        options:{
                            title : {
                                text: '手机占有率'
                            },
                            tooltip : {
                                trigger: 'item',
                                formatter: "{b}:{c}"
                            },
                            legend: {
                                data:[]
                            },
                            toolbox: {
                                show : true,
                                feature : {
                                    mark : {show: true},
                                    dataZoom : {show: true},
                                    dataView : {show: true, readOnly: false},
                                    restore : {show: true},
                                    saveAsImage : {show: true}
                                }
                            },
                            series:[{
                                name:'手机占有矩形图',
                                type:'treemap',
                                itemStyle: {
                                    normal: {
                                        label: {
                                            show: true,
                                            formatter: "{b}:{c}"
                                        },
                                        borderWidth: 1
                                    },
                                    emphasis: {
                                        label: {
                                            show: true
                                        }
                                    }
                                },
                                data:[]
                            }]
                        }
                    });
                    avalon.scan();

                    fb.requestJson({
                        remoteUrl:"./model/treemap1.json",
                        localUrl:"./model/treemap1.json",
                        success:function(data){
                            var options=vm.options;
                            var legend=[];//图例
                            var result=data.rows;
                            for(var i=0;i<result.length;i++){//获取图例数据
                                if(legend.indexOf(result[i].name)<0){
                                    legend.push(result[i].name);
                                }
                            }
                            options.legend.data=legend;
                            options.series[0].data=result;
                            var myChart = echarts.init(document.getElementById('treemap'));
                            myChart.setOption(options);
                        }
                    });
                });

            </script>
        </body>
        </html>

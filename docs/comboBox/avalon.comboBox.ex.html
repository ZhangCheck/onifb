                    <script>
                           require(["./comboBox/avalon.comboBox","./mmRequest/mmRequest"],function(){
                            var vm= avalon.define({
                                $id:"demo",
                                $comboBoxOpts:{
                                    list:[],//选择框中的内容值
                                    selectedValue:""//input中默认值
                                }
                            });

                            avalon.ajax({
                                url:"model/list.json",
                                type:"post",
                                dataType:"json",
                                contentType : "application/x-www-form-urlencoded; charset=UTF-8",
                                success:function(data){
                                     avalon.vmodels["comboBox"].selectedValue=data.defaultValue.value;
                                     avalon.vmodels["comboBox"].selectedCode=data.defaultValue.code;
                                     avalon.vmodels["comboBox"].list=data.data;
                                }

                            });
                            avalon.scan();
                        });

                    </script>

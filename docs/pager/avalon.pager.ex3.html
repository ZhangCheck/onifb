改变每页显示的数量

<!DOCTYPE html>
<html>
    <head>
        <title>分页栏组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../require.js">

        </script>
        <script>
            require.config({
                baseUrl:'../',
                paths:{
                    avalon: 'avalon.shim',
                    echarts: 'echarts',
                    domReady:'domReady',
                    fb:"avalon.fb"
                },
                map:{
                    '*':{
                        'css':'require.css',
                        'text':'require.text'
                    }
                },
                shim:{
                    avalon: { exports: "avalon" },
                    echarts:{exports:"echarts"}
                }
            });

            require(["pager/avalon.pager"], function() {
                avalon.define("test", function(vm) {
                    vm.pager = {
                        totalItems: 150,
                        perPages: 5
                    }
                    vm.perPages = 5
                    vm.$watch("perPages", function(a) {
                        var widget = avalon.vmodels.pp
                        if (widget) {
                            widget.perPages = a
                        }
                    })
                    vm.$skipArray = ["pager"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body ms-controller="test">
        <h1>改变每页显示的数量</h1>
        <div ms-widget="pager, pp"></div>
        <select ms-duplex="perPages">
            <option>5</option>
            <option>10</option>
            <option>15</option>
        </select>
    </body>
</html>

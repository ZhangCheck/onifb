<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
	<style>
		*[ms-controller],.ms-controller{
			display:none;
		}
	</style>
    <script src="../../../../../oniui/require.js"></script> //这是使用requireJs,起始项目中已经设置好，不用关心
    <script>
        require.config(...);                           //这句是amd的配置，所有项目都是一样的，在起始项目中已经设置好，不用关心
        require(["avalon.fb"],function(){             //这句是加载基础资源，起始项目中已经设置好，不用关心
                   var DEFAULT_VALUE = "江苏飞博"    //定义初始化常量
		    var vm = avalon.define({          //定义一个作用域 (vm)
                                $id:"mainCtl",       //指定作用域名称（对应dom)
				$defaultValue:DEFAULT_VALUE,    //定义非监控的变量
                                name:DEFAULT_VALUE,            //定义被监控的变量
				reset:function(){              //定义函数或事件处理函数
					vm.name = vm.$defaultValue
				}
                     })
                     avalon.scan()
        })
    </script>
</head>
<body ms-controller="mainCtl">
    <div class="mtLg mlLg"> 欢迎来到：{{name}}</div>  //使用作用域中的变量
    <input class="mtLg mlLg" type="text" ms-duplex="name"> //通过双向绑定更新变量
    <span ms-css-cursor="{{name==$defaultValue?'':'pointer'}}" ms-css-color="{{name==$defaultValue?'#BBBBBB':''}}" ms-click="reset"> 重置 </span> //通过事件更新变量、动态样式
</body>
